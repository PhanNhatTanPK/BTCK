CREATE DATABASE MyClass;
USE MyClass;

CREATE TABLE IF NOT EXISTS Account
(
	userName varchar(50) NOT null UNIQUE,
    userPass varchar(50) NOT null,
    fullName varchar(50),
    birthday date,
    sex varchar(15),
    countryside varchar(100),
    address varchar(100),
    phone varchar(15),
    email varchar(50),
    facebook varchar(100),
    instagram varchar(100),
    image varchar(100),
    note varchar(1000),
    
    PRIMARY KEY(userName)
) ENGINE = InnoDB;

CREATE TABLE IF NOT EXISTS Plan
(
    Id int,
	userName varchar(50) NOT null,
    title varchar(50),
    note varchar(1000),
    file varchar(100),
    
    PRIMARY KEY(Id),
    CONSTRAINT FK_Acc_Plan FOREIGN KEY (userName) REFERENCES Account(userName)
) ENGINE = InnoDB;
